// OPTION 2: Using bundletool with debug keystore (auto-signed)

// 1. Build the app bundle (AAB)

// 2. Generate APKs from the bundle (uses debug keystore automatically)
"C:/Program Files/Android/Android Studio/jbr/bin/java" -jar "C:/Users/medwa/Path/bundletool.jar" build-apks --bundle=build/app/outputs/bundle/release/app-release.aab --output=build/app/outputs/bundle/release/app.apks --ks=C:/Users/medwa/.android/debug.keystore --ks-pass=pass:android --ks-key-alias=androiddebugkey --key-pass=pass:android

// 3. Install to connected device (will automatically select appropriate APKs)
"C:/Program Files/Android/Android Studio/jbr/bin/java" -jar "C:/Users/medwa/Path/bundletool.jar" install-apks --apks=build/app/outputs/bundle/release/app.apks --adb="C:/Users/medwa/AppData/Local/Android/Sdk/platform-tools/adb.exe"
